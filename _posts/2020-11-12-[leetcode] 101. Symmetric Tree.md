[leetcode 101번 문제](https://leetcode.com/problems/symmetric-tree/)

문제 이해 
* 트리가 대칭인지 아닌지 판별하기 

```
class Solution:
    def isSymmetric(self, root: TreeNode) -> bool:
        def Lfirst(r, l):
            if r is not None:
                l.append(r.val)
                Lfirst(r.left, l)
                Lfirst(r.right, l)
            else:
                l.append('null')
            return l
        def Rfirst(r, ri):
            if r is not None:
                ri.append(r.val)
                Rfirst(r.right, ri)
                Rfirst(r.left, ri)
            else:
                ri.append('null')
            return ri
        
        if Lfirst(root, []) == Rfirst(root, []):
            return True
        else:
            return False
```
